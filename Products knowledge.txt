        function openPDF(product) {
            // Define your Google Drive file IDs for each product
            const fileIds = {
                'taurolock': '1j6hP5TvMvTxxJZD1K2xEuNqmyxncPjrZ',
                'exitpro': '1yryOXyfDKgy-Dqc-XzNMaGFM99v0fQOp',

                'diafix': '1oYNcg01Dcwv_1Y5IXx8WVHUBiNGcJ9cr'
            };
            
            const fileId = fileIds[product];
            if (fileId) {
                window.open(`https://drive.google.com/file/d/${fileId}/view`, '_blank');
            }
        }
        
        // Handle iframe loading errors and show placeholder
        document.addEventListener('DOMContentLoaded', function() {
            const iframes = document.querySelectorAll('.pdf-thumbnail iframe');
            iframes.forEach(iframe => {
                iframe.onerror = function() {
                    this.style.display = 'none';
                    this.nextElementSibling.style.display = 'block';
                };
            });
        });
